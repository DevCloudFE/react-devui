# en-US

Adjust the size by setting `dSize` to `larger` and `smaller`.

# zh-Hant

通过设置 `dSize` 为 `larger` `smaller` 调整尺寸。

```tsx
import { useState } from 'react';

import { DTreeSelect, DRadio } from '@react-devui/ui';

const list = [
  {
    label: 'Item 1',
    value: 'Item 1',
    children: Array(3)
      .fill(0)
      .map((item, index1) => ({
        label: `Item 1-${index1}`,
        value: `Item 1-${index1}`,
        children:
          index1 !== 2
            ? Array(5)
                .fill(0)
                .map((item, index2) => ({
                  label: `Item 1-${index1}-${index2}`,
                  value: `Item 1-${index1}-${index2}`,
                  disabled: index2 === 3,
                }))
            : undefined,
      })),
  },
  {
    label: 'Item 2',
    value: 'Item 2',
    children: [],
  },
  {
    label: 'Item 3',
    value: 'Item 3',
    children: Array(10)
      .fill(0)
      .map((item, index) => ({
        label: `Item 3-${index}`,
        value: `Item 3-${index}`,
        disabled: index === 3,
      })),
  },
];
export default function Demo() {
  const [_size, setSize] = useState('default');
  const [select, setSelect] = useState(['Item 1-0-0', 'Item 3-0']);

  const size = _size === 'default' ? undefined : _size;

  return (
    <>
      <DRadio.Group
        dList={['smaller', 'default', 'larger'].map((size) => ({
          label: size,
          value: size,
        }))}
        dModel={_size}
        dType="outline"
        onModelChange={setSize}
      />
      <br />
      <DTreeSelect dList={list} dPlaceholder="Basic" dSize={size}></DTreeSelect>
      <DTreeSelect
        dList={list}
        dPlaceholder="Multiple"
        dModel={select}
        dMultiple
        dSearchable
        dClearable
        dSize={size}
        onModelChange={setSelect}
      ></DTreeSelect>
    </>
  );
}
```
