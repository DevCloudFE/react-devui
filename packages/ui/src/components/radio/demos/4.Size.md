# en-US

Adjust the button size by setting `dSize` to `larger` and `smaller`.

# zh-Hant

通过设置 `dSize` 为 `larger` `smaller` 调整尺寸。

```tsx
import { useState } from 'react';

import { DRadio, DRadioGroup } from '@react-devui/ui';

export default function Demo() {
  const [value, setValue] = useState(2);

  return (
    <>
      <DRadioGroup dModel={[value, setValue]} dSize="smaller">
        {[1, 2, 3].map((n) => (
          <DRadio key={n} dValue={n}>
            Radio {n}
          </DRadio>
        ))}
      </DRadioGroup>
      <br />
      <DRadioGroup dModel={[value, setValue]}>
        {[1, 2, 3].map((n) => (
          <DRadio key={n} dValue={n}>
            Radio {n}
          </DRadio>
        ))}
      </DRadioGroup>
      <br />
      <DRadioGroup dModel={[value, setValue]} dSize="larger">
        {[1, 2, 3].map((n) => (
          <DRadio key={n} dValue={n}>
            Radio {n}
          </DRadio>
        ))}
      </DRadioGroup>
      <br />
      <DRadioGroup dModel={[value, setValue]} dType="outline" dSize="smaller">
        {[1, 2, 3].map((n) => (
          <DRadio key={n} dValue={n}>
            Radio {n}
          </DRadio>
        ))}
      </DRadioGroup>
      <br />
      <DRadioGroup dModel={[value, setValue]} dType="outline">
        {[1, 2, 3].map((n) => (
          <DRadio key={n} dValue={n}>
            Radio {n}
          </DRadio>
        ))}
      </DRadioGroup>
      <br />
      <DRadioGroup dModel={[value, setValue]} dType="outline" dSize="larger">
        {[1, 2, 3].map((n) => (
          <DRadio key={n} dValue={n}>
            Radio {n}
          </DRadio>
        ))}
      </DRadioGroup>
    </>
  );
}
```
