# en-US

Customize the icon through `dIcon`.

# zh-CN

通过 `dIcon` 自定义图标。

```tsx
import { QuestionOutlined } from '@react-devui/icons';
import { ToastService, DButton } from '@react-devui/ui';

export default function Demo() {
  return (
    <DButton
      onClick={() => {
        ToastService.open({
          dIcon: <QuestionOutlined />,
          dContent: 'Alert message',
        });
      }}
    >
      Open Toast
    </DButton>
  );
}
```
