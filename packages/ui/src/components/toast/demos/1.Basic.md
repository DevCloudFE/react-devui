# zh-CN

最简单的用法。

```tsx
import { ToastService, DButton } from '@react-devui/ui';

export default function Demo() {
  return (
    <DButton
      onClick={() => {
        ToastService.open({
          dContent: 'Alert message',
        });
      }}
    >
      Open Toast
    </DButton>
  );
}
```
