# zh-Hant

最简单的用法。

```tsx
import { useState } from 'react';

import { DTabs } from '@react-devui/ui';

export default function Demo() {
  const [active, setActive] = useState(22);

  return (
    <DTabs
      dTabs={[1, 22, 333, 4444].map((n) => ({
        id: n,
        title: `Tab${n}`,
        panel: `Tab${n} Content...`,
        disabled: n === 333,
      }))}
      dActive={[active, setActive]}
    />
  );
}
```
