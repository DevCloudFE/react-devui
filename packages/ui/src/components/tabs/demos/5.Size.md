# en-US

Adjust the header size by setting `dSize` to `larger` and `smaller`.

# zh-Hant

通过设置 `dSize` 为 `larger` `smaller` 调整页头尺寸。

```tsx
import { useState } from 'react';

import { DTabs, DRadioGroup } from '@react-devui/ui';

const tabs = [1, 22, 333, 4444].map((n) => ({
  id: n,
  title: `Tab${n}`,
  panel: `Tab${n} Content...`,
  disabled: n === 333,
}));
export default function Demo() {
  const [placement, setPlacement] = useState('top');
  const [_size, setSize] = useState('default');

  const size = _size === 'default' ? undefined : _size;

  return (
    <>
      <DRadioGroup
        dOptions={['top', 'right', 'bottom', 'left'].map((placement) => ({
          label: placement,
          value: placement,
        }))}
        dModel={placement}
        dType="outline"
        onModelChange={setPlacement}
      />
      <br />
      <DRadioGroup
        dOptions={['smaller', 'default', 'larger'].map((size) => ({
          label: size,
          value: size,
        }))}
        dModel={_size}
        dType="outline"
        onModelChange={setSize}
      />
      <br />
      <DTabs dTabs={tabs} dPlacement={placement} dSize={size} />
      <br />
      <DTabs dTabs={tabs} dPlacement={placement} dSize={size} dType="slider" />
      <br />
      <DTabs dTabs={tabs} dPlacement={placement} dSize={size} dType="wrap" />
    </>
  );
}
```
