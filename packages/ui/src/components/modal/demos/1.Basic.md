# zh-Hant

最简单的用法。

```tsx
import { useCallback } from 'react';

import { DModal, DButton, DModalHeader, DModalFooter, DIcon } from '@react-devui/ui';
import { useAsync, useImmer } from '@react-devui/ui/hooks';

export default function Demo() {
  const [visible, setVisible] = useImmer(false);
  const asyncCapture = useAsync();

  const handleClick = useCallback(() => {
    setVisible(true);
  }, [setVisible]);

  const handleOk = useCallback(() => {
    const [asyncGroup] = asyncCapture.createGroup('handleOk');

    return new Promise((r) => {
      asyncGroup.setTimeout(() => {
        r(true);
      }, 3000);
    });
  }, [asyncCapture]);

  return (
    <>
      <DButton onClick={handleClick}>Show Modal</DButton>
      <DModal
        dVisible={[visible, setVisible]}
        dHeader={<DModalHeader dExtraIcons={[<DIcon dName="full-screen"></DIcon>]}>Basic Drawer</DModalHeader>}
        dFooter={<DModalFooter onOkClick={handleOk}></DModalFooter>}
      >
        <div>Some contents...</div>
      </DModal>
    </>
  );
}
```
