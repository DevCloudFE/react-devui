# en-US

Realize complex footer operations through the `DModal.Footer` component.

# zh-Hant

通过 `DModal.Footer` 组件实现复杂的页脚操作。

```tsx
import { useState, useRef } from 'react';

import { useAsync } from '@react-devui/hooks';
import { DModal, DButton } from '@react-devui/ui';

export default function Demo() {
  const dataRef = useRef({});
  const async = useAsync();

  const [visible, setVisible] = useState(false);

  return (
    <>
      <DButton
        onClick={() => {
          setVisible(true);
        }}
      >
        Show Modal
      </DButton>
      <DModal
        dVisible={visible}
        dHeader="Custom Footer"
        dFooter={
          <DModal.Footer
            dAlign="center"
            dActions={[
              'cancel',
              <DButton
                onClick={() => {
                  setVisible(false);
                }}
              >
                Submit
              </DButton>,
              'ok',
            ]}
            dCancelProps={{ disabled: true }}
            onOkClick={() => {
              return new Promise((r) => {
                dataRef.current.clearTid?.();
                dataRef.current.clearTid = async.setTimeout(() => {
                  r(true);
                }, 3000);
              });
            }}
          ></DModal.Footer>
        }
        onVisibleChange={setVisible}
      >
        <div className="app-demo-container">
          <div>Some contents...</div>
          <div>Some contents...</div>
          <div>Some contents...</div>
        </div>
      </DModal>
    </>
  );
}
```
