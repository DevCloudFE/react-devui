# en-US

This example shows dynamically loading data when a node is expanded.

# zh-Hant

该示例展示了展开节点时动态加载数据。

```tsx
import { useImmer, useAsync } from '@react-devui/hooks';
import { DTree } from '@react-devui/ui';

let n = 0;
export default function Demo() {
  const asyncCapture = useAsync();

  const [list, setList] = useImmer([
    {
      value: n,
      label: 'Expand to load',
      children: [],
    },
  ]);

  return (
    <DTree
      dList={list}
      onFirstExpand={(value) => {
        n += 1;
        const findItem = (item) => {
          if (item.value === value) {
            return item;
          }
          return findItem(item.children[0]);
        };
        setList((draft) => {
          const item = findItem(draft[0]);
          item.loading = true;
        });
        asyncCapture.setTimeout(() => {
          setList((draft) => {
            const item = findItem(draft[0]);
            item.loading = false;
            item.children = [
              {
                value: n,
                label: 'Expand to load',
                children: [],
              },
            ];
          });
        }, 1000);
      }}
    />
  );
}
```
