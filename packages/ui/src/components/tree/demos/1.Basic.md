# zh-Hant

最简单的用法。

```tsx
import { useState } from 'react';

import { DTree, DSwitch } from '@react-devui/ui';

const list = [
  {
    label: 'Item 1',
    value: 'Item 1',
    children: Array(3)
      .fill(0)
      .map((item, index1) => ({
        label: `Item 1-${index1}`,
        value: `Item 1-${index1}`,
        children:
          index1 !== 2
            ? Array(5)
                .fill(0)
                .map((item, index2) => ({
                  label: `Item 1-${index1}-${index2}`,
                  value: `Item 1-${index1}-${index2}`,
                  disabled: index2 === 3,
                }))
            : undefined,
      })),
  },
  {
    label: 'Item 2',
    value: 'Item 2',
    children: [],
  },
  {
    label: 'Item 3',
    value: 'Item 3',
    children: Array(10)
      .fill(0)
      .map((item, index) => ({
        label: `Item 3-${index}`,
        value: `Item 3-${index}`,
        disabled: index === 3,
      })),
  },
];
export default function Demo() {
  const [disabled, setDisabled] = useState(false);

  return (
    <>
      <DSwitch dModel={disabled} onModelChange={setDisabled}>
        Disabled
      </DSwitch>
      <br />
      <br />
      <DTree dList={list} dDisabled={disabled} />
    </>
  );
}
```
