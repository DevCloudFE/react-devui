# en-US

Select non-leaf nodes.

# zh-Hant

选择非叶子节点。

```tsx
import { DTree } from '@react-devui/ui';

const list = [
  {
    label: 'Item 1',
    value: 'Item 1',
    children: Array(3)
      .fill(0)
      .map((item, index1) => ({
        label: `Item 1-${index1}`,
        value: `Item 1-${index1}`,
        children:
          index1 !== 2
            ? Array(3)
                .fill(0)
                .map((item, index2) => ({
                  label: `Item 1-${index1}-${index2}`,
                  value: `Item 1-${index1}-${index2}`,
                  disabled: index2 === 1,
                }))
            : undefined,
      })),
  },
  {
    label: 'Item 2',
    value: 'Item 2',
    children: [],
  },
  {
    label: 'Item 3',
    value: 'Item 3',
    children: Array(5)
      .fill(0)
      .map((item, index) => ({
        label: `Item 3-${index}`,
        value: `Item 3-${index}`,
        disabled: index === 3,
      })),
  },
];
export default function Demo() {
  return <DTree dList={list} dOnlyLeafSelectable={false} dExpandAll />;
}
```
