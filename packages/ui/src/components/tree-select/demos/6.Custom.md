# en-US

Customize selects and items.

# zh-CN

自定义已选项和选项。

```tsx
import { useState } from 'react';

import { FolderOpenOutlined, FolderOutlined, FileOutlined } from '@react-devui/icons';
import { DTreeSelect, DSwitch } from '@react-devui/ui';

const list = [
  {
    label: 'Item 1',
    value: 'Item 1',
    children: Array(3)
      .fill(0)
      .map((item, index1) => ({
        label: `Item 1-${index1}`,
        value: `Item 1-${index1}`,
        children:
          index1 !== 2
            ? Array(5)
                .fill(0)
                .map((item, index2) => ({
                  label: `Item 1-${index1}-${index2}`,
                  value: `Item 1-${index1}-${index2}`,
                  disabled: index2 === 3,
                }))
            : undefined,
      })),
  },
  {
    label: 'Item 2',
    value: 'Item 2',
    children: [],
  },
  {
    label: 'Item 3',
    value: 'Item 3',
    children: Array(10)
      .fill(0)
      .map((item, index) => ({
        label: `Item 3-${index}`,
        value: `Item 3-${index}`,
        disabled: index === 3,
      })),
  },
];
export default function Demo() {
  const [showLine, setShowLine] = useState(true);
  const [expands, setExpands] = useState(['Item 1']);

  return (
    <>
      <DSwitch dModel={showLine} onModelChange={setShowLine}>
        Show line
      </DSwitch>
      <br />
      <br />
      <DTreeSelect
        style={{ width: 200 }}
        dList={list}
        dExpands={expands}
        dPlaceholder="Customize"
        dCustomSelected={(select) => `# ${select.label}`}
        dCustomItem={(item) => (
          <div style={{ display: 'flex', alignItems: 'center', gap: '0 8px' }}>
            {item.children ? expands.includes(item.value) ? <FolderOpenOutlined /> : <FolderOutlined /> : <FileOutlined />}
            {item.label}
          </div>
        )}
        dShowLine={showLine}
        onExpandsChange={setExpands}
      ></DTreeSelect>
      <DTreeSelect
        style={{ width: 400 }}
        dList={list}
        dExpands={expands}
        dPlaceholder="Customize in multiple mode"
        dCustomSelected={(select) => `# ${select.label}`}
        dCustomItem={(item) => (
          <div style={{ display: 'flex', alignItems: 'center', gap: '0 8px' }}>
            {item.children ? expands.includes(item.value) ? <FolderOpenOutlined /> : <FolderOutlined /> : <FileOutlined />}
            {item.label}
          </div>
        )}
        dShowLine={showLine}
        onExpandsChange={setExpands}
        dMultiple
      ></DTreeSelect>
    </>
  );
}
```
