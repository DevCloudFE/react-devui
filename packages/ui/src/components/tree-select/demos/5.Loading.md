# en-US

This example shows loading on open, loading on expand, and loading on search.

# zh-Hant

该示例展示了打开时加载、展开时加载以及搜索时加载。

```tsx
import { useState } from 'react';

import { DTreeSelect } from '@react-devui/ui';
import { useImmer, useAsync } from '@react-devui/ui/hooks';

let n = 0;
export default function Demo() {
  const asyncCapture = useAsync();

  const [visible1, setVisible1] = useState(false);
  const [loading1, setLoading1] = useState(false);
  const [list1, setList1] = useState([]);

  const [list2, setList2] = useImmer([
    {
      value: n,
      label: 'Expand to load',
      children: [],
    },
  ]);

  const [loading3, setLoading3] = useState(false);
  const [list3, setList3] = useState([]);
  const [select3, setSelect3] = useState(null);

  return (
    <>
      <DTreeSelect
        dList={list1}
        dPlaceholder="Open"
        dClearable
        dVisible={visible1}
        dLoading={loading1}
        onVisibleChange={(visible) => {
          setVisible1(visible);
          if (visible && list1.length === 0) {
            setLoading1(true);
            asyncCapture.setTimeout(() => {
              setLoading1(false);
              setList1([
                {
                  label: 'Item 1',
                  value: 'Item 1',
                  children: Array(3)
                    .fill(0)
                    .map((item, index1) => ({
                      label: `Item 1-${index1}`,
                      value: `Item 1-${index1}`,
                      children:
                        index1 !== 2
                          ? Array(5)
                              .fill(0)
                              .map((item, index2) => ({
                                label: `Item 1-${index1}-${index2}`,
                                value: `Item 1-${index1}-${index2}`,
                                disabled: index2 === 3,
                              }))
                          : undefined,
                    })),
                },
                {
                  label: 'Item 2',
                  value: 'Item 2',
                  children: [],
                },
                {
                  label: 'Item 3',
                  value: 'Item 3',
                  children: Array(10)
                    .fill(0)
                    .map((item, index) => ({
                      label: `Item 3-${index}`,
                      value: `Item 3-${index}`,
                      disabled: index === 3,
                    })),
                },
              ]);
            }, 1000);
          }
        }}
      ></DTreeSelect>
      <DTreeSelect
        dList={list2}
        dPlaceholder="Expand"
        dClearable
        onFirstExpand={(value) => {
          n += 1;
          const findItem = (item) => {
            if (item.value === value) {
              return item;
            }
            return findItem(item.children[0]);
          };
          setList2((draft) => {
            const item = findItem(draft[0]);
            item.loading = true;
          });
          asyncCapture.setTimeout(() => {
            setList2((draft) => {
              const item = findItem(draft[0]);
              item.loading = false;
              item.children = [
                {
                  value: n,
                  label: 'Expand to load',
                  children: [],
                },
              ];
            });
          }, 1000);
        }}
      ></DTreeSelect>
      <DTreeSelect
        dList={list3}
        dModel={select3}
        dPlaceholder="Search"
        dClearable
        dSearchable
        dLoading={loading3}
        onSearch={(value) => {
          const [asyncGroup] = asyncCapture.createGroup('Search');
          setSelect3(null);
          if (value.length === 0) {
            setLoading3(false);
            setList3([]);
          } else {
            setLoading3(true);
            setList3([]);
            asyncGroup.setTimeout(() => {
              setLoading3(false);
              setList3([
                {
                  label: 'Item 1',
                  value: 'Item 1',
                  children: Array(3)
                    .fill(0)
                    .map((item, index1) => ({
                      label: `Item 1-${index1}`,
                      value: `Item 1-${index1}`,
                      children:
                        index1 !== 2
                          ? Array(5)
                              .fill(0)
                              .map((item, index2) => ({
                                label: `Item 1-${index1}-${index2}`,
                                value: `Item 1-${index1}-${index2}`,
                                disabled: index2 === 3,
                              }))
                          : undefined,
                    })),
                },
                {
                  label: 'Item 2',
                  value: 'Item 2',
                  children: [],
                },
                {
                  label: 'Item 3',
                  value: 'Item 3',
                  children: Array(10)
                    .fill(0)
                    .map((item, index) => ({
                      label: `Item 3-${index}`,
                      value: `Item 3-${index}`,
                      disabled: index === 3,
                    })),
                },
              ]);
            }, 1000);
          }
        }}
        onModelChange={setSelect3}
      ></DTreeSelect>
    </>
  );
}
```
