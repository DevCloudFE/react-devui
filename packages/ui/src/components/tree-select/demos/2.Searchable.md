# en-US

Search for options.

The search can be customized through `dCustomSearch`.

# zh-Hant

对选项进行搜索。

可通过 `dCustomSearch` 自定义搜索。

```tsx
import { DTreeSelect } from '@react-devui/ui';

const list = [
  {
    label: 'Item 1',
    value: 'Item 1',
    children: Array(3)
      .fill(0)
      .map((item, index1) => ({
        label: `Item 1-${index1}`,
        value: `Item 1-${index1}`,
        children:
          index1 !== 2
            ? Array(5)
                .fill(0)
                .map((item, index2) => ({
                  label: `Item 1-${index1}-${index2}`,
                  value: `Item 1-${index1}-${index2}`,
                  disabled: index2 === 3,
                }))
            : undefined,
      })),
  },
  {
    label: 'Item 2',
    value: 'Item 2',
    children: [],
  },
  {
    label: 'Item 3',
    value: 'Item 3',
    children: Array(10)
      .fill(0)
      .map((item, index) => ({
        label: `Item 3-${index}`,
        value: `Item 3-${index}`,
        disabled: index === 3,
      })),
  },
];
const search = {
  filter: (value, item) => {
    return item.label.endsWith(value);
  },
  sort: () => -1,
};
export default function Demo() {
  return (
    <>
      <DTreeSelect dList={list} dPlaceholder="Search" dSearchable></DTreeSelect>
      <DTreeSelect dList={list} dPlaceholder="Custom search" dSearchable dCustomSearch={search}></DTreeSelect>
    </>
  );
}
```
