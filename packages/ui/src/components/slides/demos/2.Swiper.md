# zh-Hant

更多用法参考 [Swiper](https://swiperjs.com/react)。

```tsx
import { DSlides } from '@react-devui/ui';

export default function Demo() {
  return (
    <DSlides
      style={{ height: 200 }}
      direction={'vertical'}
      pagination={{
        dynamicBullets: true,
      }}
    >
      {Array(8)
        .fill(0)
        .map((n, i) => (
          <DSlides.Slide key={i}>
            <div className="app-demo-slide">Slide {i}</div>
          </DSlides.Slide>
        ))}
    </DSlides>
  );
}
```
