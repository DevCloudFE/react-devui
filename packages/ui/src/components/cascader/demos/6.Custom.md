# en-US

Customize selects and items.

# zh-Hant

自定义已选项和选项。

```tsx
import { DCascader } from '@react-devui/ui';
import { InfoCircleOutlined } from '@react-devui/ui/icons';

const list = Array(3)
  .fill(0)
  .map((item, index1) => ({
    label: `Item ${index1}`,
    value: `${index1}`,
    children: Array(5)
      .fill(0)
      .map((item, index2) => ({
        label: `Item ${index1}-${index2}`,
        value: `${index1}-${index2}`,
        disabled: index2 === 3,
        children: Array(20)
          .fill(0)
          .map((item, index3) => ({
            label: `Item ${index1}-${index2}-${index3}`,
            value: `${index1}-${index2}-${index3}`,
          })),
      })),
  }));
export default function Demo() {
  return (
    <>
      <DCascader
        dList={list}
        dPlaceholder="Customize"
        dCustomSelected={(select) => `# ${select.label}`}
        dCustomItem={(item) => (
          <div style={{ display: 'flex', alignItems: 'center', gap: '0 8px' }}>
            {item.label}
            <InfoCircleOutlined />
          </div>
        )}
      ></DCascader>
      <DCascader
        dList={list}
        dPlaceholder="Customize in multiple mode"
        dCustomSelected={(select) => `# ${select.label}`}
        dCustomItem={(item) => (
          <div style={{ display: 'flex', alignItems: 'center', gap: '0 8px' }}>
            {item.label}
            <InfoCircleOutlined />
          </div>
        )}
        dMultiple
      ></DCascader>
    </>
  );
}
```
