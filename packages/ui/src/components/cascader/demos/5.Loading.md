# en-US

This example shows loading on open, loading on focus, and loading on search.

# zh-Hant

该示例展示了打开时加载、聚焦时加载以及搜索时加载。

```tsx
import { useState } from 'react';

import { DCascader } from '@react-devui/ui';
import { useImmer, useAsync } from '@react-devui/ui/hooks';

export default function Demo() {
  const asyncCapture = useAsync();

  const [visible1, setVisible1] = useState(false);
  const [loading1, setLoading1] = useState(false);
  const [options1, setOptions1] = useState([]);

  const [options2, setOptions2] = useState(
    Array(3)
      .fill(0)
      .map((item, index1) => ({
        dLabel: `Option ${index1 + 1}`,
        dValue: index1 + 1,
        dChildren: [],
      }))
  );

  const [loading3, setLoading3] = useState(false);
  const [options3, setOptions3] = useState([]);
  const [select3, setSelect3] = useState(null);

  return (
    <div className="app-demo-flex-container">
      <DCascader
        style={{ width: 240 }}
        dVisible={[visible1, setVisible1]}
        dOptions={options1}
        dPlaceholder="Open"
        dLoading={loading1}
        dClearable
        onVisibleChange={(visible) => {
          const [asyncGroup] = asyncCapture.createGroup('Open');
          if (visible && options1.length === 0) {
            setLoading1(true);
            asyncGroup.setTimeout(() => {
              setLoading1(false);
              setOptions1(
                Array(3)
                  .fill(0)
                  .map((item, index1) => ({
                    dLabel: `Option ${index1 + 1}`,
                    dValue: index1 + 1,
                    dChildren: Array(5)
                      .fill(0)
                      .map((item, index2) => ({
                        dLabel: `Option ${index1 + 1}${index2 + 1}`,
                        dValue: index2 + 1,
                        dDisabled: index2 === 3,
                        dChildren: Array(5)
                          .fill(0)
                          .map((item, index3) => ({
                            dLabel: `Option ${index1 + 1}${index2 + 1}${index3 + 1}`,
                            dValue: index3 + 1,
                          })),
                      })),
                  }))
              );
            }, 1000);
          }
        }}
      ></DCascader>
      <DCascader
        style={{ width: 240 }}
        dOptions={options2}
        dPlaceholder="Focus"
        dClearable
        onFocusChange={(value) => {
          const [asyncGroup] = asyncCapture.createGroup('Focus');
          if (value.length === 1) {
            const index1 = options2.findIndex((item) => item.dValue === value[0]);
            if (options2[index1].dLoading === undefined) {
              options2[index1].dLoading = true;
              setOptions2([...options2]);
              asyncGroup.setTimeout(() => {
                options2[index1].dLoading = false;
                options2[index1].dChildren = Array(5)
                  .fill(0)
                  .map((item, index2) => ({
                    dLabel: `Option ${index1 + 1}${index2 + 1}`,
                    dValue: index2 + 1,
                    dDisabled: index2 === 3,
                    dChildren: Array(5)
                      .fill(0)
                      .map((item, index3) => ({
                        dLabel: `Option ${index1 + 1}${index2 + 1}${index3 + 1}`,
                        dValue: index3 + 1,
                      })),
                  }));
                setOptions2([...options2]);
              }, 1000);
            }
          }
        }}
      ></DCascader>
      <DCascader
        style={{ width: 240 }}
        dModel={[select3, setSelect3]}
        dOptions={options3}
        dPlaceholder="Search"
        dLoading={loading3}
        dClearable
        dSearchable
        onSearch={(value) => {
          const [asyncGroup] = asyncCapture.createGroup('Search');
          setSelect3(null);
          if (value.length === 0) {
            setLoading3(false);
            setOptions3([]);
          } else {
            setLoading3(true);
            setOptions3([]);
            asyncGroup.setTimeout(() => {
              setLoading3(false);
              setOptions3(
                Array(3)
                  .fill(0)
                  .map((item, index1) => ({
                    dLabel: `${value} ${index1 + 1}`,
                    dValue: index1 + 1,
                    dChildren: Array(5)
                      .fill(0)
                      .map((item, index2) => ({
                        dLabel: `${value} ${index1 + 1}${index2 + 1}`,
                        dValue: index2 + 1,
                        dDisabled: index2 === 3,
                        dChildren: Array(5)
                          .fill(0)
                          .map((item, index3) => ({
                            dLabel: `${value} ${index1 + 1}${index2 + 1}${index3 + 1}`,
                            dValue: index3 + 1,
                          })),
                      })),
                  }))
              );
            }, 1000);
          }
        }}
      ></DCascader>
    </div>
  );
}
```
