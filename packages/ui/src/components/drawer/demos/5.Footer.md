# en-US

Realize complex footer operations through the `DDrawerFooter` component.

# zh-Hant

通过 `DDrawerFooter` 组件实现复杂的页脚操作。

```tsx
import { useCallback } from 'react';
import { useImmer } from 'use-immer';

import { DDrawer, DDrawerHeader, DDrawerFooter, DButton, DIcon } from '@react-devui/ui';
import { useAsync } from '@react-devui/ui/hooks';

export default function Demo() {
  const asyncCapture = useAsync();

  const [visible, setVisible] = useImmer(false);

  const handleClick = useCallback(() => {
    setVisible(true);
  }, [setVisible]);

  const handleClose = useCallback(() => {
    setVisible(false);
  }, [setVisible]);

  const handleOk = useCallback(() => {
    const [asyncGroup] = asyncCapture.createGroup('handleOk');

    return new Promise((r) => {
      asyncGroup.setTimeout(() => {
        r(true);
      }, 3000);
    });
  }, [asyncCapture]);

  const submit = <DButton onClick={handleClose}>Submit</DButton>;

  return (
    <>
      <DButton onClick={handleClick}>Show Drawer</DButton>
      <DDrawer
        dVisible={visible}
        dHeader={<DDrawerHeader>Custom Footer</DDrawerHeader>}
        dFooter={
          <DDrawerFooter
            dAlign="center"
            dButtons={['cancel', submit, 'ok']}
            dCancelButtonProps={{ disabled: true }}
            onOkClick={handleOk}
          ></DDrawerFooter>
        }
        onClose={handleClose}
      >
        <p>Some contents...</p>
        <p>Some contents...</p>
        <p>Some contents...</p>
      </DDrawer>
    </>
  );
}
```
