# en-US

Set the container through `dContainer`.

# zh-Hant

通过 `dContainer` 设置容器。

```tsx
import { useCallback } from 'react';
import { useImmer } from 'use-immer';

import { DDrawer, DButton } from '@react-devui/ui';

export default function Demo() {
  const [visible, setVisible] = useImmer(false);

  const handleClick = useCallback(() => {
    setVisible(true);
  }, [setVisible]);

  const handleClose = useCallback(() => {
    setVisible(false);
  }, [setVisible]);

  return (
    <div className="container">
      <DButton onClick={handleClick}>Show Drawer</DButton>
      <DDrawer dVisible={visible} dContainer={false} onClose={handleClose}>
        <p>Some contents...</p>
        <p>Some contents...</p>
        <p>Some contents...</p>
      </DDrawer>
    </div>
  );
}
```

```scss
.container {
  height: 200px;
  padding: 20px;
  background-color: rgba(var(--d-color-primary-rgb) / 10%);
  border: 1px solid var(--d-divider-color);
}
```
