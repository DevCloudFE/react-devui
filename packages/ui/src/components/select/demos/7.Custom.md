# en-US

Customize selects and options.

# zh-Hant

自定义已选项和选项。

```tsx
import { DSelect } from '@react-devui/ui';
import { InfoCircleOutlined } from '@react-devui/ui/icons';

const options = Array(100)
  .fill(0)
  .map((item, index) => ({
    label: `Option ${index}`,
    value: index,
    disabled: index === 3,
  }));
export default function Demo() {
  return (
    <>
      <DSelect
        dOptions={options}
        dPlaceholder="Customize"
        dCustomSelected={(select) => `# ${select.label}`}
        dCustomOption={(option) => (
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
            {option.label}
            <InfoCircleOutlined />
          </div>
        )}
      />
      <DSelect
        dOptions={options}
        dPlaceholder="Customize in multiple mode"
        dCustomSelected={(select) => `# ${select.label}`}
        dCustomOption={(option) => (
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
            {option.label}
            <InfoCircleOutlined />
          </div>
        )}
        dMultiple
      />
    </>
  );
}
```
