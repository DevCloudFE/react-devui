# en-US

The form supports asynchronous validation.

# zh-Hant

表单支持异步校验。

```tsx
import { DForm, DFormItem, FormControl, FormGroup, useForm, DInput } from '@react-devui/ui';

const asyncValidatorFn = (control) => {
  return new Promise((r) => {
    setTimeout(() => {
      if (control.value.length > 5) {
        r({ maxLength: true });
      } else if (control.value.length > 0) {
        r(null);
      } else {
        r({ required: true });
      }
    }, 1000);
  });
};

export default function Demo() {
  const formInstance = useForm(
    () =>
      new FormGroup({
        username: new FormControl('', [], asyncValidatorFn),
      })
  );

  return (
    <DForm dForm={formInstance} dLabelWidth={120} dFeedbackIcon>
      <DFormItem
        dFormControls={{
          username: {
            required: 'Please input your username!',
            maxLength: {
              message: 'Maximum length is 5!',
              status: 'warning',
            },
          },
        }}
        dLabel="Username"
      >
        {({ username }) => <DInput dFormControl={username} placeholder="Asynchronous verification" />}
      </DFormItem>
    </DForm>
  );
}
```
