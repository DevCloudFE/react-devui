# zh-Hant

这里展示了所有受支持的组件。

```tsx
import { useCallback, useState } from 'react';

import {
  DForm,
  DFormItem,
  FormControl,
  FormGroup,
  useForm,
  DInput,
  DInputAffix,
  DButton,
  DRadio,
  DRadioGroup,
  DSelect,
  DTextarea,
  DCheckbox,
  DCheckboxGroup,
  DSwitch,
  DCascader,
  DSlider,
} from '@react-devui/ui';

export default function Demo() {
  const [options] = useState(() =>
    Array(100)
      .fill(0)
      .map((item, index) => ({ dLabel: `Option ${index + 1}`, dValue: index + 1, dDisabled: index === 3 }))
  );
  const [cascaderOptions] = useState(() =>
    Array(3)
      .fill(0)
      .map((item, index1) => ({
        dLabel: `Option ${index1 + 1}`,
        dValue: index1 + 1,
        dChildren: Array(5)
          .fill(0)
          .map((item, index2) => ({
            dLabel: `Option ${index1 + 1}${index2 + 1}`,
            dValue: index2 + 1,
            dDisabled: index2 === 3,
            dChildren: Array(20)
              .fill(0)
              .map((item, index3) => ({
                dLabel: `Option ${index1 + 1}${index2 + 1}${index3 + 1}`,
                dValue: index3 + 1,
              })),
          })),
      }))
  );
  const formInstance = useForm(
    () =>
      new FormGroup({
        Input: new FormControl('', () => ({ error: true })),
        Number: new FormControl('', () => ({ error: true })),
        Checkbox: new FormControl(true),
        CheckboxGroup: new FormControl([1]),
        Radio: new FormControl(true),
        RadioGroup: new FormControl(1),
        Select: new FormControl(50, () => ({ error: true })),
        Cascader: new FormControl([1, 1, 10], () => ({ error: true })),
        Textarea: new FormControl('', () => ({ error: true })),
        Switch: new FormControl(false),
        Slider: new FormControl(40, () => ({ error: true })),
      })
  );

  const handleVerify = useCallback(
    (n) => {
      Object.values(formInstance.form.controls).forEach((control) => {
        control.markAsDirty();
      });
      formInstance.updateForm();
    },
    [formInstance]
  );

  return (
    <DForm style={{ minWidth: 600 }} dForm={formInstance}>
      <DFormItem dLabel="Input" dErrors="Errors!">
        <DInput dFormControlName="Input" />
      </DFormItem>
      <DFormItem dLabel="NumberInput" dErrors="Errors!">
        <DInputAffix dNumber>
          <DInput dFormControlName="Number" placeholder="Number" />
        </DInputAffix>
      </DFormItem>
      <DFormItem dLabel="Checkbox">
        <DCheckbox dFormControlName="Checkbox">Checkbox</DCheckbox>
      </DFormItem>
      <DFormItem dLabel="CheckboxGroup">
        <DCheckboxGroup dFormControlName="CheckboxGroup">
          {[1, 2, 3].map((n) => (
            <DCheckbox key={n} dValue={n} dDisabled={n === 2}>
              Checkbox {n}
            </DCheckbox>
          ))}
        </DCheckboxGroup>
      </DFormItem>
      <DFormItem dLabel="Radio">
        <DRadio dFormControlName="Radio">Radio</DRadio>
      </DFormItem>
      <DFormItem dLabel="RadioGroup">
        <DRadioGroup dFormControlName="RadioGroup">
          {[1, 2, 3].map((n) => (
            <DRadio key={n} dValue={n} dDisabled={n === 2}>
              Radio {n}
            </DRadio>
          ))}
        </DRadioGroup>
      </DFormItem>
      <DFormItem dLabel="Switch">
        <DSwitch dFormControlName="Switch"></DSwitch>
      </DFormItem>
      <DFormItem dLabel="Select" dErrors="Errors!">
        <DSelect dFormControlName="Select" dOptions={options}></DSelect>
      </DFormItem>
      <DFormItem dLabel="Cascader" dErrors="Errors!">
        <DCascader dFormControlName="Cascader" dOptions={cascaderOptions}></DCascader>
      </DFormItem>
      <DFormItem dLabel="Textarea" dErrors="Errors!">
        <DTextarea dFormControlName="Textarea" dRows="auto" />
      </DFormItem>
      <DFormItem dLabel="Slider" dErrors="Errors!">
        <DSlider dFormControlName="Slider"></DSlider>
      </DFormItem>
      <DFormItem>
        <DButton onClick={handleVerify}>Verify</DButton>
      </DFormItem>
    </DForm>
  );
}
```
