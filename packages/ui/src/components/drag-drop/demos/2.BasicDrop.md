# en-US

The most basic drop.

# zh-Hant

最基本的放置。

```tsx
import { useImmer } from 'use-immer';

import { DDrag, DDrop, DDragPlaceholder, DButtonGroup, DButton } from '@react-devui/ui';

export default function Demo() {
  const [data, setData] = useImmer([1, 2, 3]);

  return (
    <>
      <DButtonGroup>
        <DButton
          onClick={() =>
            setData((draft) => {
              draft.unshift(draft.length + 1);
            })
          }
        >
          Add top
        </DButton>
        <DButton
          onClick={() =>
            setData((draft) => {
              draft.splice(~~(draft.length / 2), 0, draft.length + 1);
            })
          }
        >
          Add center
        </DButton>
        <DButton
          onClick={() =>
            setData((draft) => {
              draft.push(draft.length + 1);
            })
          }
        >
          Add end
        </DButton>
        <DButton onClick={() => setData([1, 2, 3])}>Clear</DButton>
      </DButtonGroup>
      <br />
      <br />
      <DDrop className="app-demo-drop" dPlaceholder={<DDragPlaceholder className="app-demo-drag-placeholder" />}>
        {data.map((n) => (
          <DDrag key={n} dId={String(n)}>
            <div className="app-demo-drag">Drag {n}</div>
          </DDrag>
        ))}
      </DDrop>
    </>
  );
}
```
