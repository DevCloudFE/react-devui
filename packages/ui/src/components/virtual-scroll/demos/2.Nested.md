# en-US

Nested lists are supported.

# zh-Hant

支持嵌套列表。

```tsx
import { useMemo } from 'react';

import { DVirtualScroll } from '@react-devui/ui';

export default function Demo() {
  const vsPerformance = useMemo(
    () => ({
      dList: [
        {
          label: 'Group 1',
          children: Array(20)
            .fill(0)
            .map((item, index) => ({
              label: `Item1 ${index}`,
            })),
        },
        {
          label: 'Group 2',
          children: [],
        },
        {
          label: 'Group 3',
          children: Array(100000)
            .fill(0)
            .map((item, index) => ({
              label: `Item3 ${index}`,
            })),
        },
      ],
      dItemSize: 32,
      dItemNested: (item) => ({
        list: item.children,
        emptySize: 32,
        asItem: false,
      }),
      dItemKey: (item) => item.label,
    }),
    []
  );

  return (
    <DVirtualScroll
      {...vsPerformance}
      dFillNode={<li></li>}
      dItemRender={(item, index, { iARIA, iChildren }) => {
        if (item.children) {
          return (
            <ul key={item.label} role="group">
              <li key={item.label} style={{ position: 'relative', left: -32, height: 32 }} role="presentation">
                {item.label}
              </li>
              {iChildren}
            </ul>
          );
        }

        return (
          <li {...iARIA} key={item.label} style={{ height: 32 }}>
            {item.label}
          </li>
        );
      }}
      dSize={192}
      dEmptyRender={() => <li style={{ height: 32 }}>No Data</li>}
    >
      {({ vsScrollRef, vsRender, vsOnScroll }) => (
        <ul ref={vsScrollRef} style={{ height: 192, overflow: 'auto' }} onScroll={vsOnScroll}>
          {vsRender}
        </ul>
      )}
    </DVirtualScroll>
  );
}
```
