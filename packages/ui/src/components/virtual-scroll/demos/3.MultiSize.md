# en-US

Support multi-size mixing.

# zh-Hant

支持多尺寸混合。

```tsx
import { DVirtualScroll } from '@react-devui/ui';

export default function Demo() {
  const list = Array(10000)
    .fill()
    .map((item, index) => 'Item ' + (index + 1));

  return (
    <DVirtualScroll
      dHeight={200}
      dList={list}
      dRenderItem={(item, index) => (
        <div key={item} className={index % 2 ? 'item-container' : undefined} style={{ height: index % 2 ? 40 : 60 }}>
          {item}
        </div>
      )}
      dCustomSize={(item, index) => (index % 2 ? 40 : 60)}
    ></DVirtualScroll>
  );
}
```

```scss
.item-container {
  background-color: rgba(var(--d-color-primary-rgb) / 10%);
}
```
