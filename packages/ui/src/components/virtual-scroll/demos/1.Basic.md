# zh-Hant

最简单的用法。

```tsx
import { DVirtualScroll } from '@react-devui/ui';

export default function Demo() {
  const list = Array(10000)
    .fill()
    .map((item, index) => 'Item ' + (index + 1));

  return (
    <DVirtualScroll
      dHeight={200}
      dList={list}
      dRenderItem={(item) => (
        <div key={item} style={{ height: 50 }}>
          {item}
        </div>
      )}
    ></DVirtualScroll>
  );
}
```
