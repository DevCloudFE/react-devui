# en-US

Support horizontal scrolling.

# zh-Hant

支持水平滚动。

```tsx
import { DVirtualScroll } from '@react-devui/ui';

export default function Demo() {
  const list = Array(10000)
    .fill()
    .map((item, index) => 'Item ' + (index + 1));

  return (
    <DVirtualScroll
      dWidth="100%"
      dList={list}
      dRenderItem={(item) => (
        <div key={item} style={{ display: 'inline-block', width: 100 }}>
          {item}
        </div>
      )}
    ></DVirtualScroll>
  );
}
```
