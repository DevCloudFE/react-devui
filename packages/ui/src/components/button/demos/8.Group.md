# en-US

Put multiple buttons as a group into the button group container.

# zh-Hant

将多个按钮作为一组放入按钮组容器中。

```tsx
import { DButtonGroup, DButton, DIcon } from '@react-devui/ui';

export default function Demo() {
  return (
    <>
      <DButtonGroup>
        <DButton>L</DButton>
        <DButton>M</DButton>
        <DButton>R</DButton>
      </DButtonGroup>
      <DButtonGroup>
        <DButton dShape="round">L</DButton>
        <DButton>M</DButton>
        <DButton dShape="round">R</DButton>
      </DButtonGroup>
      <DButtonGroup>
        <DButton dType="primary">L</DButton>
        <DButton>M</DButton>
        <DButton dType="dashed">R</DButton>
      </DButtonGroup>
      <br />
      <br />
      <DButtonGroup dType="outline">
        <DButton>L</DButton>
        <DButton>M</DButton>
        <DButton>R</DButton>
      </DButtonGroup>
      <DButtonGroup dColor="danger">
        <DButton>L</DButton>
        <DButton>M</DButton>
        <DButton>R</DButton>
      </DButtonGroup>
      <DButtonGroup dSize="larger">
        <DButton>L</DButton>
        <DButton>M</DButton>
        <DButton>R</DButton>
      </DButtonGroup>
    </>
  );
}
```
