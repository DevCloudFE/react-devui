# zh-CN

最简单的用法。

```tsx
import { DTooltip, DButton } from '@react-devui/ui';

export default function Demo() {
  return (
    <div className="container">
      <DTooltip dTitle="Hover">
        <DButton>Hover</DButton>
      </DTooltip>
      <DTooltip dTitle="Click" dTrigger="click">
        <DButton>Click</DButton>
      </DTooltip>
    </div>
  );
}
```
