# en-US

Group list.

# zh-Hant

对选项进行分组。

```tsx
import { useState } from 'react';

import { DAutoComplete, DInput } from '@react-devui/ui';

export default function Demo() {
  const [value, setValue] = useState('');
  const [list, setList] = useState([]);

  return (
    <DAutoComplete
      dList={list}
      onItemClick={(val) => {
        setValue(val);
      }}
    >
      <DInput
        dModel={value}
        onModelChange={(val) => {
          setValue(val);
          setList(
            Array(val ? 3 : 0)
              .fill(0)
              .map((n, i) => ({
                value: `Group ${i + 1}`,
                children: Array(3)
                  .fill(0)
                  .map((cn, ci) => ({
                    value:
                      `G${i + 1}-` +
                      Array(ci + 1)
                        .fill(val)
                        .join(''),
                  })),
              }))
          );
        }}
      />
    </DAutoComplete>
  );
}
```
