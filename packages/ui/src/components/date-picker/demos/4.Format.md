# en-US

Formatted display through `dFormat`.

# zh-Hant

通过 `dFormat` 格式化显示。

```tsx
import { DDatePicker } from '@react-devui/ui';

export default function Demo() {
  return (
    <>
      <DDatePicker dFormat="DD/MM/YYYY" />
      <DDatePicker dShowTime dFormat="YYYY/MM/DD HH:mm" />
    </>
  );
}
```
